atLength :: ([a] -> b) -> (Int -> b) -> [a] -> Int -> b

atLength atLen atEnd ls n unravels list ls to position n. Precisely: 

  atLength atLenPred atEndPred ls n
   | n < 0         = atLenPred n
   | length ls < n = atEndPred (n - length ls)
   | otherwise     = atLenPred (drop n ls)
